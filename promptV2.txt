1. Missing F3X:ValidateTool() function body
The F3X:ValidateTool() method is defined but never actually implemented — it just calls notify() and equipBuildingTool() but equipBuildingTool is a global function defined later, not a method. This will error when called during builds.
2. F3XRetry wrapper is broken
lua
local function F3XRetry(method, ...)
    for retry = 1, 3 do
        local result = F3X[method](F3X, ...)
This calls F3X[method] but method is a string key like "SyncMove" — however F3X:SyncMove() etc. are defined as methods with colon syntax (:), so F3X["SyncMove"] returns the function but calling it as F3X[method](F3X, ...) is correct. However, F3XRetry is never actually used consistently — some places use F3X:SyncMove() directly, others use F3XRetry. The prompt requires every F3X call to go through retry logic.

4. Hyper Mode coroutine tracking is broken
lua
for t = 1, tc do threads[t] = task.spawn(function() buildChunk(chunks[t]) end) end
for t = 1, tc do while coroutine.status(threads[t]) ~= "dead" do task.wait(0.1) end end
task.spawn() returns a thread, but coroutine.status() expects a coroutine. task.spawn creates a new thread, not a coroutine. This will error. Use task.wait() with a completion counter instead.
5. Missing applyPartProperties SyncMove call
In applyPartProperties, you have:
lua
if partData.Size then table.insert(szC, {Part = part, Size = ..., CFrame = targetCF}) end
But the prompt says SyncMove must be called immediately after CreatePart with the target CFrame, before any other ops. Your applyPartProperties calls SyncMove inside a batch with size changes mixed in, and the initial SyncMove in the build loop happens before applyPartProperties. This is actually mostly correct, but the szC table incorrectly includes CFrame in SyncResize when it should only be in SyncMove.
6. SyncRotate is defined but never used properly
The prompt lists SyncRotate for orientation adjustments, but your script only uses it if partData.Orientation exists (which is non-standard). Standard builds use CFrame for rotation.

This is the "copy and paste in place" feature you mentioned:
lua
-- Add to Export tab or as a new "Duplicate" button:
local duplicateBtn = Instance.new("TextButton")
duplicateBtn.Text = "📋 Smart Duplicate"
duplicateBtn.Size = UDim2.new(1, 0, 0, 30)
duplicateBtn.Position = UDim2.fromOffset(0, 205) -- adjust
duplicateBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 200)
duplicateBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
duplicateBtn.TextSize = 14
duplicateBtn.Font = Enum.Font.GothamBold
duplicateBtn.BorderSizePixel = 0
Instance.new("UICorner", duplicateBtn).CornerRadius = UDim.new(0, 6)
duplicateBtn.Parent = exportTab

duplicateBtn.MouseButton1Click:Connect(function()
    local parts = {}
    for part, _ in pairs(State.Selection) do
        if part and part.Parent then table.insert(parts, part) end
    end
    if #parts == 0 then notify("No parts selected"); return end
    
    -- Calculate offset (e.g., duplicate to the right by bounding box width)
    local bounds = {}
    for _, part in ipairs(parts) do
        table.insert(bounds, part.Position + part.Size / 2)
        table.insert(bounds, part.Position - part.Size / 2)
    end
    
    local minX, maxX = math.huge, -math.huge
    for _, pos in ipairs(bounds) do
        minX = math.min(minX, pos.X)
        maxX = math.max(maxX, pos.X)
    end
    local offset = Vector3.new(maxX - minX + 2, 0, 0) -- 2 stud gap
    
    -- Clone via F3X
    local cloned = F3XRetry("Clone", parts)
    if not cloned then notify("Clone failed"); return end
    
    -- Move cloned parts by offset
    local moveChanges = {}
    for i, part in ipairs(cloned) do
        table.insert(moveChanges, {
            Part = part,
            CFrame = part.CFrame + offset
        })
    end
    F3XRetry("SyncMove", moveChanges)
    
    notify("Duplicated " .. #parts .. " parts with offset")
    
    -- Select the new parts
    SelectionSystem:Clear()
    for _, part in ipairs(cloned) do
        SelectionSystem:SelectPart(part)
    end
end)

most optimized way to do Ghost Preview — designed for minimal lag with 10K+ parts:
Core Strategy: Bounding Box + Sample Parts
Instead of creating ghost copies of every part, you show only the outline and a few sample parts so the user can see scale and orientation without lag.
lua
local GhostSystem = {}

-- Cache
local ghostFolder = nil
local ghostParts = {}      -- sparse table: ghostParts[partIndex] = ghostPart
local ghostAdornments = {} -- selection boxes for all parts (cheaper than parts)
local ghostBounds = nil    -- single part showing overall bounds
local ghostEnabled = false

-- Constants
local GHOST_LIMIT = 200        -- max actual ghost parts
local GHOST_COLOR = Color3.fromRGB(0, 170, 255)
local GHOST_TRANSPARENCY = 0.85
local GHOST_MATERIAL = Enum.Material.ForceField

function GhostSystem.Clear()
    if ghostFolder then
        pcall(function() ghostFolder:Destroy() end)
        ghostFolder = nil
    end
    ghostParts = {}
    ghostAdornments = {}
    ghostBounds = nil
    ghostEnabled = false
end

function GhostSystem.Update(buildData)
    GhostSystem.Clear()
    if not buildData or not buildData.Parts then return end
    
    ghostEnabled = true
    ghostFolder = Instance.new("Folder")
    ghostFolder.Name = "F3X_Ghost"
    ghostFolder.Parent = workspace
    
    local total = #buildData.Parts
    local sampleRate = math.max(1, math.floor(total / GHOST_LIMIT))
    
    -- Calculate bounds while sampling
    local minPos = Vector3.new(math.huge, math.huge, math.huge)
    local maxPos = Vector3.new(-math.huge, -math.huge, -math.huge)
    
    for i, pd in ipairs(buildData.Parts) do
        local cf = parseCFrame(pd.CFrame)
        local size = parseVector3(pd.Size)
        
        -- Apply offset/rotation/scale from CONFIG
        if CONFIG.BuildOffset and CONFIG.BuildOffset.Magnitude > 0 then
            cf = cf + CONFIG.BuildOffset
        end
        if CONFIG.BuildRot90 and CONFIG.BuildRot90 ~= 0 then
            cf = cf * CFrame.Angles(0, math.rad(CONFIG.BuildRot90), 0)
        end
        if CONFIG.BuildScale and CONFIG.BuildScale ~= 1 then
            size = size * CONFIG.BuildScale
        end
        
        -- Update bounds
        local half = size / 2
        local pos = cf.Position
        minPos = Vector3.new(
            math.min(minPos.X, pos.X - half.X),
            math.min(minPos.Y, pos.Y - half.Y),
            math.min(minPos.Z, pos.Z - half.Z)
        )
        maxPos = Vector3.new(
            math.max(maxPos.X, pos.X + half.X),
            math.max(maxPos.Y, pos.Y + half.Y),
            math.max(maxPos.Z, pos.Z + half.Z)
        )
        
        -- Sample: create ghost part every Nth part
        if i % sampleRate == 0 or i == 1 or i == total then
            local ghost = Instance.new("Part")
            ghost.Name = "Ghost_" .. i
            ghost.CFrame = cf
            ghost.Size = size
            ghost.Anchored = true
            ghost.CanCollide = false
            ghost.CanQuery = false
            ghost.CanTouch = false
            ghost.Transparency = GHOST_TRANSPARENCY
            ghost.Color = GHOST_COLOR
            ghost.Material = GHOST_MATERIAL
            ghost.CastShadow = false
            ghost.Parent = ghostFolder
            
            -- Add selection box for wireframe effect
            local box = Instance.new("SelectionBox")
            box.Adornee = ghost
            box.Color3 = GHOST_COLOR
            box.LineThickness = 0.02
            box.Parent = ghost
            
            ghostParts[i] = ghost
        end
    end
    
    -- Create bounds box
    local center = (minPos + maxPos) / 2
    local boundsSize = maxPos - minPos
    
    ghostBounds = Instance.new("Part")
    ghostBounds.Name = "Ghost_Bounds"
    ghostBounds.CFrame = CFrame.new(center)
    ghostBounds.Size = boundsSize
    ghostBounds.Anchored = true
    ghostBounds.CanCollide = false
    ghostBounds.CanQuery = false
    ghostBounds.Transparency = 0.95
    ghostBounds.Color = Color3.fromRGB(255, 100, 0)
    ghostBounds.Material = Enum.Material.Neon
    ghostBounds.CastShadow = false
    ghostBounds.Parent = ghostFolder
    
    -- Wireframe on bounds
    local boundsBox = Instance.new("SelectionBox")
    boundsBox.Adornee = ghostBounds
    boundsBox.Color3 = Color3.fromRGB(255, 100, 0)
    boundsBox.LineThickness = 0.03
    boundsBox.Parent = ghostBounds
    
    notify(string.format("👻 Ghost: %d sample parts + bounds (%d total)", 
        math.min(total, GHOST_LIMIT), total))
end

-- Debounced update for slider changes
local updateQueued = false
function GhostSystem.QueueUpdate(buildData)
    if updateQueued then return end
    updateQueued = true
    task.delay(0.15, function()
        updateQueued = false
        if ghostEnabled and buildData then
            GhostSystem.Update(buildData)
        end
    end)
end
